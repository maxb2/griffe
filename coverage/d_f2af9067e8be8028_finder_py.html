316def _handle_pth_file(path) -> list[Path]: # noqa: WPS231 

317 # Support for .pth files pointing to directories. 

318 # From https://docs.python.org/3/library/site.html: 

319 # A path configuration file is a file whose name has the form name.pth 

320 # and exists in one of the four directories mentioned above; 

321 # its contents are additional items (one per line) to be added to sys.path. 

322 # Non-existing items are never added to sys.path, 

323 # and no check is made that the item refers to a directory rather than a file. 

324 # No item is added to sys.path more than once. 

325 # Blank lines and lines beginning with # are skipped. 

326 # Lines starting with import (followed by space or tab) are executed. 

327 directories = [] 

328 for line in path.read_text(encoding="utf8").strip().splitlines(keepends=False): 

329 line = line.strip() 

330 if line and not line.startswith("#") and not _re_import_line.search(line): 

331 if os.path.exists(line): 

332 directories.append(Path(line)) 

333 return directories
